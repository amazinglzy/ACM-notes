#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <cmath>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
const int P = 1000000007;
LL f[1000001], v[1000001];
LL rp(LL now, int k)
{
    LL will = 1;
    for (; k; k >>= 1, now *= now, now %= P)
    {
        if (k & 1) will *= now, will %= P;
    }
    return will;
}
LL C(int n, int m)
{
    if(n < m) return 0;
    return f[n] * rp(f[m], P - 2) % P * rp(f[n - m], P - 2) % P;
}
void init()
{
    f[0] = 1; v[0] = 1;
    for (int i = 1; i <= 1000000; i++) //1e6以内的组合数
    {
        f[i] = f[i - 1] * i % P;
    }
}
int main()
{
    init();
    int n, m;
    scanf("%d %d", &n, &m);
    printf("%lld\n", C(m, n));
}
